"""Module for interaction with the Snok service container"""

from typing import List
from task_runner.plugin import Plugin
from task_runner.parameters import Param
from conx_cli_tools.lib.docker_runner import docker_exec


CONTAINER_NAME = "snok"

plugin = Plugin(help="Wrapper for snok")



[docs]@plugin.task(
    title="Prepare HPA and ITA-HIL environment for Snok",
    help="""
    1. Set Platform Version
    2. Push TCPIP Configuration to HPA
    3. Generate Service Specification given commit-id
""")
async def setup_snok(
    commit_id: str = Param(..., "[Required] cs-software-product commit-id (git hash)"),
    platform_version: str = Param(..., "[Required] CSP Platform version (e.g. 2210.5.3)")
):
    assert platform_version
    assert commit_id

    await docker_exec(
        container_name=CONTAINER_NAME,
        command=['setup_config', '--non-interactive'],
        env_vars={
            'CSP_PLATFORM_VERSION': str(platform_version)
        },
        working_dir='/snok_config',
        execute_as_str=True
    )

    await docker_exec(
        container_name=CONTAINER_NAME,
        command=['push_config'],
        working_dir='/snok_config',
        execute_as_str=True,
    )

    await docker_exec(
        container_name=CONTAINER_NAME,
        command=['generate_service_spec', 'install', commit_id,
                 '--output-dir', '/snok_config/', '--force'],
        execute_as_str=True
    )




[docs]@plugin.task(title="Apply TCPIP Runtime for ALL Services")
async def apply_tcpip_runtime(
        reverse: bool = Param(
            False, "[Optional] Apply IPC Runtime for ALL Services"
        ),
        partial: List[str] = Param(
            [], "[Optional] Apply TCPIP on given set of services. Rest will use IPC."
        )):

    cmd = ['apply_tcpip_runtime']
    if reverse and partial:
        raise Exception('Cannot use --reverse and --partial at the same time')

    if reverse:
        cmd.extend(['--reverse'])
    if partial:
        cmd.append('--partial')
        cmd.extend(partial)

    await docker_exec(
        container_name=CONTAINER_NAME,
        command=cmd,
        working_dir='/snok_config',
        execute_as_str=True
    )
