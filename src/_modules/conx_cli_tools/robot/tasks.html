"""
Tasks for robot tests
"""
import logging
import os
from task_runner.plugin import Plugin
from task_runner.parameters import Param
from conx_cli_tools.lib import docker_runner
from conx_cli_tools.lib.utils import convert_str_to_dict, get_task_id_from_task_name
from conx_cli_tools.robot import update_vars_file
from conx_cli_tools.azure_sync.tasks import status, get_dest


LOGGER = logging.getLogger(__name__)
LOGGER.setLevel(logging.INFO)
plugin = Plugin(help="Wrapper for robot tests")

async def run_tests(
    robot_arguments: str = Param(
        '',
        "The arguments for robot framework"
    ),
    robot_sources: str = Param(
        ...,
        "Relative path from test_repo_folder to folder with robot test files or relative path to robot test files"
    ),
    test_repo_folder: str = Param(
        os.getcwd(),
        "The absolute path to the test repository root folder (default current working directory)"
    ),
    env_vars: str = Param(
        '',
        ("The environment variables in format 'env_key=env_value"
         " env_another_key=env_another_value ...'")
    ),
    report_results: str = Param(
        'false',
        "Report results to victoria (Used when running in chain)"
    ),
    robot_vars_file: str = Param(
        '',
        "Relative path from test_repo_folder to a robot vars file. Leave empty for no update"
    )
) -> None:
    """Run tests"""
    cmd_list = ['robot']
    robot_arguments_list = robot_arguments.split()
    env_variables = {}

    robot_arguments_list.extend(['--pythonpath', test_repo_folder])

    result_folder = os.environ.get('result_folder', '')

    if result_folder:
        robot_arguments_list.extend(["--outputdir", os.path.join(result_folder, 'robot_client')])

    if report_results.lower() == 'true':
        robot_arguments_list.extend(['--listener', 'visualization'])
        task_id = await get_task_id_from_task_name('run_tests')
        env_variables = {
            'activity_id': os.environ.get('activity_id', ''),
            'job_name': os.environ.get('job_name', ''),
            'task_id': task_id
        }
        azure_sync_status_response = await status()

        LOGGER.info(f"Azure sync status {azure_sync_status_response}")

        azure_destination = ''
        for azure_sync_obj in azure_sync_status_response:
            if get_dest() in azure_sync_obj['destination']:
                azure_destination = azure_sync_obj['destination']

        LOGGER.info(f"Azure destination {azure_destination}")
        if azure_destination:
            env_variables['log_link'] = f"{azure_destination}/robot_client/log.html"
            env_variables['report_link'] = f"{azure_destination}/robot_client/report.html"

    if env_vars:
        env_variables = {**env_variables, **convert_str_to_dict(env_vars)}

    if robot_vars_file and os.path.exists(os.path.join(test_repo_folder, robot_vars_file)):
        update_vars_file.load_vars(robot_vars_file)
        robot_arguments_list.extend(['--variablefile', robot_vars_file])

    cmd_list.extend(robot_arguments_list)
    cmd_list.append(robot_sources)
    await docker_runner.docker_exec(
        container_name='robot_client',
        command=cmd_list,
        env_vars=env_variables,
        working_dir=test_repo_folder
    )
