CONFIG_ALIARO_IP = '192.168.1.149'

[docs]def load_vars(path_to_vars_file):
    """
    Load the variables from file
    """
    with open(path_to_vars_file, 'r') as stream:
        try:
            variables = yaml.load(stream, Loader=yaml.Loader)
        except yaml.YAMLError as exc:
            print(exc)
            variables = {}

    if variables:
        variables['DIAGNOSTICS'] = {}
        variables['DIAGNOSTICS']['IP'] = os.environ.get(
            'resource_metadata_config_sga_dlt_ip', CONFIG_SGA_DLT_IP)
        variables['ETH_SWITCH_IP'] = os.environ.get(
            'resource_metadata_config_eth_switch_ip', CONFIG_ETH_SWITCH_IP)
        variables['CANOE']['CONFIG_DIRECTORY'] = ''
        variables['ALIARO']['ALIARO_IP'] = os.environ.get(
            'resource_metadata_config_aliaro_ip', CONFIG_ALIARO_IP)
        physical_ecus = os.environ.get(
            'resource_metadata_physical_ecus', PHYSICAL_ECUS)
        variables['PHYSICAL_ECUS'] = physical_ecus if isinstance(physical_ecus, list) else physical_ecus.split()
        print(physical_ecus)
        variables['SERIAL_PORTS']['DHUH_COM_PORT'] = {}
        variables['SERIAL_PORTS']['DHUH_COM_PORT']['port'] = os.environ.get(
            'resource_metadata_config_dhuh_com_port', CONFIG_DHUH_COM_PORT)
        variables['SERIAL_PORTS']['DHUM_COM_PORT'] = {}
        variables['SERIAL_PORTS']['DHUM_COM_PORT']['port'] = os.environ.get(
            'resource_metadata_config_dhum_com_port', CONFIG_DHUM_COM_PORT)
        variables['SERIAL_PORTS']['DHUM_COM_PORT']['write'] = os.environ.get(
            'resource_metadata_config_dhum_com_write', CONFIG_DHUM_COM_PORT)

        variables['DLT_HOSTS'] = {'SGA_DLT': {}, 'HPA_DLT': {}, 'DHU_DLT': {},
                                  'TCA_DLT': {}}
        variables['DLT_HOSTS']['SGA_DLT']['ip'] = os.environ.get(
            'resource_metadata_config_sga_dlt_ip', CONFIG_SGA_DLT_IP)
        variables['DLT_HOSTS']['SGA_DLT']['port'] = os.environ.get(
            'resource_metadata_config_sga_dlt_port', CONFIG_SGA_DLT_PORT)
        variables['DLT_HOSTS']['HPA_DLT']['ip'] = os.environ.get(
            'resource_metadata_config_hpa_dlt_ip', CONFIG_HPA_DLT_IP)
        variables['DLT_HOSTS']['HPA_DLT']['port'] = os.environ.get(
            'resource_metadata_config_hpa_dlt_port', CONFIG_HPA_DLT_PORT)
        variables['DLT_HOSTS']['DHU_DLT']['ip'] = os.environ.get(
            'resource_metadata_config_dhu_dlt_ip', CONFIG_DHU_DLT_IP)
        variables['DLT_HOSTS']['DHU_DLT']['port'] = os.environ.get(
            'resource_metadata_config_dhu_dlt_port', CONFIG_DHU_DLT_PORT)
        variables['DLT_HOSTS']['TCA_DLT']['ip'] = os.environ.get(
            'resource_metadata_config_tca_dlt_ip', CONFIG_TCA_DLT_IP)
        variables['DLT_HOSTS']['TCA_DLT']['port'] = os.environ.get(
            'resource_metadata_config_tca_dlt_port', CONFIG_TCA_DLT_PORT)
        variables['LOG_FOLDER'] = os.path.join(os.environ.get("result_folder", os.getcwd()), 'robot_client', 'logs')
        try:
            del variables['SERIAL_PORTS']['IHU_VIP_COM_PORT']
            del variables['SERIAL_PORTS']['IHU_MP_COM_PORT']
            del variables['SERIAL_PORTS']['TCAM_COM_PORT']
        except KeyError:
            pass

        with open(path_to_vars_file, 'w') as stream:
            yaml.dump(variables, stream=stream, Dumper=Dumper, indent=4,
                      sort_keys=False)



if __name__ == '__main__':
    try:
        file_path = sys.argv[1]

        print('You must specify path to vars.yml')
