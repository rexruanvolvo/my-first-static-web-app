"""Module connecting general commands to the azure sync"""
import json
import logging
import os
from string import Template
import requests_async as requests
from task_runner.parameters import Param
from task_runner.plugin import Plugin
from task_runner.settings import GlobalSettings

from conx_cli_tools.lib.utils import AzureCredentials


LOGGER = logging.getLogger(__name__)
LOGGER.setLevel(logging.INFO)
plugin = Plugin(help="Wrapper for Azure sync")

[docs]def get_dest():
    activity_id = os.environ.get("activity_id", "")
    job_id = os.environ.get("job_id", "")
    job_name = os.environ.get("job_name", "")
    product_name = job_name.split("/")[0]
    activity_name = ""
    task_name = ""
    if job_name.count("/") == 2:
        activity_name = job_name.split("/")[1]
        task_name = job_name.split("/")[2]
        destination_path = f'{product_name}/{activity_name}/{activity_id}/{task_name}/{job_id}'
    elif job_name.count("/") == 1:
        activity_name = job_name.split("/")[1]
        destination_path = f'{product_name}/{activity_name}/{activity_id}/{job_id}'
    else:
        destination_path = f'{product_name}/{activity_id}/{job_id}'
    return destination_path




[docs]@plugin.task(title="Azure sync start")
async def start(
    source: str = Param(
        os.environ.get("result_folder"),
        "Source Logs Path"
    ),
    dest: str = Param(
        get_dest(),
        "Destinath Logs Path"
    )
):
    azure_storage = GlobalSettings.instance.get_token("azureStorage", AzureCredentials)
    logs_url = Template(azure_storage.AzureDomainLogsStorageSAS)
    dest = logs_url.substitute(destination_path=dest)
    try:
        response = await requests.post(
            f'http://127.0.0.1:{os.environ.get("AZURE_SYNC_PORT")}/api/start',
            json=json.dumps({
                'SOURCE': source,
                'DESTINATION': dest
            })
        )
        LOGGER.info(response.json())
    except requests.exceptions.ConnectionError:
        raise TaskRunnerException("Connection error occurred when tried to start asure sync")
