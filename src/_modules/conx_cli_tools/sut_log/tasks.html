[docs]@plugin.task(title="Start logging {interfaces}")
async def start(
    output_folder: str = Param(
        os.path.join(os.environ.get('result_folder', os.getcwd()), 'log_server'),
        "Output folder"
    ),
    interfaces: str = Param(
        "dlt serial canoe ethernet",
        "Space separated strings"
    ),
    continue_on_fail: str = Param(
        "false",
        "Not raise errors when failing"
    )
):
    run_cmd = ["start_logging", "--output_folder", output_folder]
    if interfaces:
        run_cmd.append("--interfaces")
        run_cmd.extend(interfaces.split())
    await execute_log_client_command(run_cmd, continue_on_fail)
