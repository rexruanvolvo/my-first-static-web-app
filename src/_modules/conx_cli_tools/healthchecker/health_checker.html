"""
Python script to run health check on robot_client.
"""

import time
import os
from conx_cli_tools.lib.call import call, call_pipe


TIMEOUT_TIME = 15
PAUSE_TIME = 5
WAIT_TIME = 100
N_TRIES = 3
ERROR_EXEC_STR = "raising: __main__.HealthCheckerException: "
ERROR_MSG_UNHEALTHY = "docker-compose is unhealthy."
ERROR_MSG_STARTING = (
    "docker-compose state is starting, docker-compose up took too long."
)
ERROR_MSG_TIMEOUT = (
    "Could not read docker-compose status due to process.communicate() timed out."
)
ERROR_MSG_OTHER = "docker-compose is "

LOG_FILE_NAME = os.path.join(
    os.environ.get("result_folder", os.getcwd()), "healthcheck.log"
)
PREV_WORKDIR = os.getcwd()
CYNOSURE_SERVICES_PATH = os.environ.get("CYNOSURE_SERVICES_PATH")
ENVIRONMENT = os.environ.get("ENVIRONMENT", "")
CONTAINER_NAME = f'robot_client_{ENVIRONMENT}' if ENVIRONMENT else 'robot_client'

[docs]async def save_docker_logs():
    """
    Function to save docker logs
    """
    save_str_to_logs("")
    save_str_to_logs("Running docker-compose logs -t")
    await call(f"docker-compose logs -t >> {LOG_FILE_NAME}")
    save_str_to_logs(
        'Running docker inspect --format "{{json .State.Health }}" ' + CONTAINER_NAME
    )
    await call(
        'docker inspect --format "{{json .State.Health }}"'
        f" {CONTAINER_NAME} | jq . | cat >> {LOG_FILE_NAME}"
    )
    save_str_to_logs("Running docker-compose ps")
    await call(f"docker-compose ps | cat >> {LOG_FILE_NAME}")
    save_str_to_logs("")



def _format_health_string(string):
    string = str(string)
    string = string.strip()
    string = string.rstrip('"')
    string = string.lstrip('"')
    return string



[docs]async def health_check():

    health = await docker_inspect()
    loops_tot = WAIT_TIME // PAUSE_TIME
    n_loops = 0
    print(f"What was the health? It was {health}.")
    while health != "healthy" and n_loops != loops_tot:
        save_str_to_logs(
            f"Waited {n_loops*PAUSE_TIME} seconds, current status is {health}."
        )
        time.sleep(PAUSE_TIME)
        n_loops += 1
        health = await docker_inspect()
        if health in ("unhealthy", "process.communicate() timed out."):
            break
    return health




[docs]async def ensure_healthy_status():
    for i in range(N_TRIES):
        save_str_to_logs(f"Try number {i+1}/{N_TRIES}")
        health = await health_check()
        await save_docker_logs()
        if health == "healthy":
            save_str_to_logs(f"The robot client is {health}")
            break
        if health == "unhealthy":
            save_str_to_logs(
                f"{CONTAINER_NAME} is {health} after {i+1} tries.")
            if i + 1 != N_TRIES:
                save_str_to_logs("Running docker-compose restart")
                await docker_compose_restart()
    await stdout_outcome(health)




[docs]async def stdout_outcome(health):
    """
    Function return outcome to stdout
    """
    if health == "healthy":
        save_str_to_logs(f"docker-compose is {health}")
        return  # return to not raise error if healthy
    save_str_to_logs(f"Raise exception due to state: {health}")
    if health == "unhealthy":
        save_str_to_logs(ERROR_EXEC_STR + ERROR_MSG_UNHEALTHY)
        raise HealthCheckerException(ERROR_MSG_UNHEALTHY)
    if health == "starting":
        save_str_to_logs(ERROR_EXEC_STR + ERROR_MSG_STARTING)
        raise HealthCheckerException(ERROR_MSG_STARTING)
    if health == "process.communicate() timed out.":
        save_str_to_logs(ERROR_EXEC_STR + ERROR_MSG_TIMEOUT)
        raise HealthCheckerException(ERROR_MSG_TIMEOUT)
    save_str_to_logs(ERROR_EXEC_STR + ERROR_MSG_OTHER + health)
    raise HealthCheckerException(ERROR_MSG_OTHER + health)
