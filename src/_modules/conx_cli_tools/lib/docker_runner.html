import logging

LOGGER = logging.getLogger(__name__)
LOGGER.setLevel(logging.INFO)



[docs]async def docker_exec(container_name: str,
                      command: list,
                      env_vars: dict = None,
                      env_files: list = None,
                      working_dir: str = '',
                      user: str = '',
                      execute_as_str: bool = False):
    """Exec docker command"""
    cmd = ["docker", "exec", "--interactive", "--privileged"]
    if env_vars:
        for key, value in env_vars.items():
            cmd.extend(["--env", f"{key}={value}"])
    if env_files:
        for env_file in env_files:
            if os.path.exists(env_file):
                cmd.extend(["--env-file", env_file])
            else:
                LOGGER.warning(f"Environment file {env_file} does not exist...")
    if working_dir:
        cmd.extend(["--workdir", working_dir])
    if user:
        cmd.extend(["--user", user])

    container_name_with_env = container_name
    environment = os.environ.get("ENVIRONMENT", "")
    if environment:
        container_name_with_env = f"{container_name}_{environment}"
    cmd.append(container_name_with_env)
    cmd.extend(command)
    LOGGER.info(f"Executing docker command -- {' '.join(cmd)}")

    # Some commands only work if provided as str
    if execute_as_str:
        cmd = ' '.join(cmd)

    await sh(cmd=cmd)
