[docs]def convert_str_to_dict(variables: str) -> Dict[str, str]:
    """
    Converts strings on form 'key=value other_key=other_value'
    into dictionaries
    """
    dict_vars = {}
    pairs = variables.split()
    for pair in pairs:
        if '=' not in pair:
            raise ValueError('key-value pair given incorrectly (' + variables + ')')
        if pair.startswith('='):
            raise ValueError("Missing key for value '" + pair[1:] + "'")
        if pair.endswith('='):
            raise ValueError("Missing value for key '" + pair[:-1] + "'")
        key, val = pair.split('=')
        dict_vars[key] = val
    return dict_vars

[docs]async def get_task_id_from_task_name(task_name: str) -> str:
    """
    Get victoria task id from task name (Only to be used when task is executed in a chain)
    """
    if is_running_locally():
        logger.warning("Task is executed locally... No taskid to resolve from cynosure!")
        return ''
    timeout = 0
    while timeout < 30:
        tasks_data = await _get_json_data_from_file(os.path.join(os.environ.get('workspace'), 'tasks.json'))
        for task in tasks_data:
            if task['name'] == task_name:
                return task['taskId']
        timeout += 1
        await asyncio.sleep(1)
    logger.warning(f"Task with name '{task_name}' was not resolved in tasks.json!")
    return ''
