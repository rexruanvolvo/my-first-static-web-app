[docs]    async def confighub_swdl(self, args):
        """Method to call OSB to software download."""
        if not args.integration_manifest and not args.baselines:
            raise OSBHandlerException('You need to provide either integration manifest or baselines')
        baselines = []
        if args.integration_manifest:
            baselines.extend(IntegrationManifestHandler.get_baselines_from_manifest(args.integration_manifest))
        baselines.extend(args.baselines if args.baselines else [])
        baselines_str = ','.join(dict.fromkeys(baselines))
        # should we move customized testobject_config into the default path for testobject_config
        command = f"/App/OSB confighub_swdl --testobject_config {args.workspace}/test-object-config.json"
        command = f"{command} --confighub_user {args.confighub_user} --confighub_password {args.confighub_password}"
        command = f"{command} --artifactory_user {args.artifactory_username}"
        command = f"{command} --artifactory_password {args.artifactory_apikey}"
        command = f"{command} --baseline {baselines_str}"
        if args.ip_address:
            command = f"{command} --ip_address {args.ip_address}"
        if os.environ.get("result_folder"):
            command = f"{command} --log_dir {os.path.join(os.environ.get('result_folder'), 'osb')}"
        await docker_exec(
            container_name=f"{CONTAINER_NAME}",
            command=command.split()
        )
