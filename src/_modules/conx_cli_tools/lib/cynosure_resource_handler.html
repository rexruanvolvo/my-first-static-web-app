[docs]class CynosureResourceHandler:  # pylint: disable=too-few-public-methods
    """Resource class"""

    def __init__(self, resource_meta_json):
        self.resource_meta_json = resource_meta_json
        self._metadata = {}

    # pylint: disable=expression-not-assigned
    def _flatten_dictionary(self, init_dict, parent_key=""):
        """
        Flatten a given dictionary by pattern %s.%s for the keys
        :param init_dict: Raw dictionary
        :param parent_key: Parent key of the value in case of a recursive call
        :return final_dict: flattened init_dictionary
        """
        final_dict = []
        for key, value in init_dict.items():
            try:
                if parent_key:
                    final_dict.extend(self._flatten_dictionary(value, f"{parent_key}.{key}").items())
                else:
                    final_dict.extend(self._flatten_dictionary(value, key).items())
            except AttributeError:
                final_dict.append((key, value)) if not parent_key else final_dict.append((f"{parent_key}.{key}", value))
        return dict(final_dict)

    @property
    def metadata(self):
        """Property to get the metadata from resource metadata file"""
        if not self._metadata:
            if not os.path.isfile(self.resource_meta_json):
                raise CynosureResourceHandlerException(
                    f"Unable to load resource metadata, file {self.resource_meta_json} not found!"
                )
            with open(self.resource_meta_json) as f:  # pylint: disable=unspecified-encoding
                metadata_dict = json.load(f)
                self._metadata = self._flatten_dictionary(metadata_dict)
        return self._metadata
