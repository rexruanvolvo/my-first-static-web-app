"""Module containing general handler enabling use of various handlers"""
import importlib
import logging


LOGGER = logging.getLogger(__name__)
LOGGER.setLevel(logging.INFO)



[docs]class Handler():
    """General handler enabling use of various handlers"""

    def __init__(self, relay):
        if not relay:
            self.class_object = None
            print("No relay specified...")
            return
        if relay.endswith('_handler'):
            module_name = relay
        else:
            module_name = f'{relay}_handler'  # Module as string, eg. 'aliaro_handler'

        # If handler is called with "my_module_name", this line outputs "MyModuleName"
        relay_class = ''.join([word.capitalize() for word in module_name.split('_')])
        try:
            relay_module = importlib.import_module(
                f'conx_cli_tools.lib.{module_name}'
            )
            # Calling class function, eg. AliaroHandler()
            self.class_object = getattr(relay_module, f'{relay_class}')()
        except (ImportError, AttributeError):
            LOGGER.warning(
                f'No relay handler implemented... Make sure {relay}_handler.py'
                f' exists and that the relay handler class is called {relay_class}'
            )
            self.class_object = None


[docs]    def execute_function(self, func_name, **args):
        if self.class_object:
            try:
                # Executes class method, eg. AliaroHandler.power_on_ecus(**args)
                return getattr(self.class_object, func_name)(**args)
            except AttributeError:
                LOGGER.warning(f'{func_name} is not implemented')
        return None
