    """
    Integration Manifest Handler.
    """

    @staticmethod
    def _get_dict_from_yaml(file_path):
        """
        Return dictionary data from yaml file
        :param file_path: YAML file to be read
        """
        with open(file_path, 'r', encoding="utf8") as yaml_file:
            manifest_data = yaml.load(yaml_file, Loader=yaml.FullLoader)
        return manifest_data


[docs]    @staticmethod
    def get_baselines_from_manifest(file_path):
        """get baseline from manifest file"""
        baselines = []
        manifest_data = IntegrationManifestHandler._get_dict_from_yaml(file_path)
        if manifest_data.get("dependencies") and manifest_data.get("type") == "integration-manifest":
            dependencies = manifest_data["dependencies"]
            for _, dep_value in dependencies.items():
                if dep_value.get("metadata"):
                    meta_data = dep_value["metadata"]
                    if meta_data.get("baseline") and meta_data["baseline"]:
                        baseline = get_product_instance(
                            dep_value["namespace"],
                            dep_value["version"]
                        )
                        if baseline:
                            baselines.append(baseline)
        return baselines




[docs]def get_product_instance(product, version):
    """get product from DB"""
    data = {
        "query": {
            "productId": {
                "namespace": product
            },
            "version": str(version)
        },
        "limit": 1
    }
    response = requests.post(
        'https://core.statedb.cynosure.volvocars.biz/api/3.0.0/product/query',
        json=data,
        verify=False
    )
    response_json = response.json()[0]
    baseline_instance = None

    if response_json:
        baseline_instance = response_json["productId"]["instance"]
    return baseline_instance
